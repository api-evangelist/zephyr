Zephyr "Hello World" sample build and flash example.

# Initialize a workspace and update repositories
west init -m https://github.com/zephyrproject-rtos/zephyr --mr main zephyrproject
cd zephyrproject
west update

# Export Zephyr CMake package and install Python deps
west zephyr-export
pip install -r zephyr/scripts/requirements.txt

# Install the Zephyr SDK
cd ~
wget https://github.com/zephyrproject-rtos/sdk-ng/releases/download/v0.16.5/zephyr-sdk-0.16.5_linux-x86_64.tar.xz
tar xvf zephyr-sdk-0.16.5_linux-x86_64.tar.xz
cd zephyr-sdk-0.16.5
./setup.sh

# Build the hello_world sample for an nRF52840 DK
cd ~/zephyrproject
west build -p auto -b nrf52840dk/nrf52840 zephyr/samples/hello_world

# Flash to the connected board
west flash

# Open a serial console (115200 8N1) and you should see:
#   *** Booting Zephyr OS build vX.Y.Z ***
#   Hello World! nrf52840dk/nrf52840
